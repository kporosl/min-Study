# 관계 연산자 연습

```c
#include <stdio.h> 

int main () {

    int A = 10;
    int B = 20;
    int x , y , z ;

    x = (A==B);
    y = (A<=B);
    z = (A>=B);

    printf("x : %d\n" , x);
    printf("y : %d\n", y);
    printf("z : %d\n", z);
    return 0 ;

};
```
